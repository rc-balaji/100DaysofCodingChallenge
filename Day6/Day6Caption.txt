class Solution {
    public int minMovesToCaptureTheQueen(int a, int b, int c, int d, int e, int f) {
        if(a==e){
            if(a==c && (d-b)*(d-f)<0){
                return 2;
            }else{
                return 1;
            }
        }
        if(b==f){
            if(b==d &&  (c-a)*(c-e)<0){
                return 2;
            }else{
                return 1;
            }
        }

        if(c-e == d-f ){
            if(a-e==b-f  && (a-c)*(a-e)<0){
                return 2;
            }else{

                return 1;
            }
        }

        if(c-e == f-d){
            if(a-e==f-b  && (a-c)*(a-e)<0){
                return 2;
            }else{

                return 1;
            }
        }

        return 2;


    }
}
